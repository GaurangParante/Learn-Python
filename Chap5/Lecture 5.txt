# Loops in Python

-> Loops are use to repeat instruction 
-> Types of Loops
    while
    for

# While Loops
while condition:
    Some work

# Break & Continue

Break: Used to terminate the loop when encountered.
Continue: Terminarte execution in the current iteration & continues execution of the loop with the next iteration.


# For Loops
-> For loops are used for sequential traversal. 
-> For traversing list,string,tuples etc.

For Loops

list = [1,2,3,4,5,6]
for el in list:
    print(el)

For loops with else

tuple = (1,2,3,4,5,6)
for el in tuple:
    print(el)
else:
    print('End of loop')


# range()

-> Range function returns a sequence of numbers.
-> Starting with 0 by default, and increments by 1 by default, and stops before specified number. 

range(start?,stop,step?)

for i in range(5):
    print(i)

for i in range(1,5):
    print(i)

for i in range(1,5,2):
    print(i)
